> EXERCISE 1 - Estrai tutti i clienti con nome Mario

SELECT * FROM customers 
WHERE name = 'Mario'

> EXERCISE 2 - Estrarre il nome e cognome dei clienti nati nel 1982

SELECT name, surname FROM customers 
WHERE birth_year = 1982

> EXERCISE 3 - Estrarre il numero delle fatture con iva al 20%

SELECT COUNT(*) as invoices_at_20 FROM invoices 
WHERE iva = 20

> EXERCISE 4 - Estrarre i prodotti attivati nel 2017 e che sono in produzione oppure in commercio

SELECT * FROM products
WHERE EXTRACT(YEAR FROM activation_date) = 2017 
AND (in_production = true OR on_market = true)

> EXERCISE 5 - Estrarre le fatture con importo inferiore a 1000 e i dati dei clienti ad esse collegate

SELECT i.*, 
c.name AS customer_name, 
c.surname AS customer_surname,
c.birth_year AS customer_birth_year,
c.residence_region AS customer_residence_region
FROM invoices as i
JOIN customers as c on i.customer_id = c.customer_number
WHERE amount < 1000

> EXERCISE 6 - Riportare l'elenco delle fatture (numero, importo, iva e data) con in aggiunta il nome del fornitore

SELECT 
i.invoice_number,
i.amount,
i.iva,
s.name AS supplier
FROM invoices as i
JOIN suppliers as s on i.supplier_number = s.supplier_number

> EXERCISE 7 - Considerando soltanto le fatture con iva al 20 per cento, estrarre il numero di fatture (conteggio fatture) per ogni anno

SELECT EXTRACT(YEAR FROM invoice_date) as year, COUNT(*) as invoices_at_20 FROM invoices
WHERE iva = 20
GROUP BY EXTRACT(YEAR FROM invoice_date)

> EXERCISE 8 - Riportare il numero di fatture (conteggio fatture) e la somma dei relativi importi divisi per anno di fatturazione

SELECT EXTRACT(YEAR FROM invoice_date) as year, 
COUNT(*) as invoices_per_year,
SUM(invoices.amount) as total_amount_per_year
FROM invoices
GROUP BY EXTRACT(YEAR FROM invoice_date)

> EXERCISE 9 - Estrarre gli anni in cui sono state registrate più di 2 fatture con tipologia 'A'

SELECT EXTRACT(YEAR FROM invoice_date) as year, 
COUNT(*) as invoices_per_year FROM invoices
WHERE type = 'A'
GROUP BY EXTRACT(YEAR FROM invoice_date)
HAVING COUNT(*) > 2

> EXERCISE 10 - Estrarre il totale degli importi delle fatture divisi per residenza dei clienti

SELECT c.residence_region as region, 
SUM(amount) as invoices_total_amount_per_region FROM invoices as i 
JOIN customers as c on i.customer_id = c.customer_number
GROUP BY c.residence_region

> EXERCISE 11 _ Estrarre il numero dei clienti nati nel 1980 che hanno almeno una fattura superiore a  50 euro

SELECT COUNT(DISTINCT c) as customers_born_in_1980_with_invoice_amount_greater_than_50
FROM customers as c 
JOIN invoices as i on c.customer_number = i.customer_id
WHERE c.birth_year = 1980 AND i.amount > 50
