> EXERCISE 1 - Estrai tutti i clienti con nome Mario

SELECT * FROM customers 
WHERE name = 'Mario'

> EXERCISE 2 - Estrarre il nome e cognome dei clienti nati nel 1982

SELECT name, surname FROM customers 
WHERE birth_year = 1982

> EXERCISE 3 - Estrarre il numero delle fatture con iva al 20%

SELECT COUNT(iva) as invoices_at_20 FROM invoices 
WHERE iva = 20

> EXERCISE 4 - Estrarre i prodotti attivati nel 2017 e che sono in produzione oppure in commercio

SELECT * FROM products
WHERE EXTRACT(YEAR FROM activation_date) = 2017 
AND (in_production = true OR on_market = true)

> EXERCISE 5 - Estrarre le fatture con importo inferiore a 1000 e i dati dei clienti ad esse collegate

SELECT i.*, 
c.name AS customer_name, 
c.surname AS customer_surname,
c.birth_year AS customer_birth_year,
c.residence_region AS customer_residence_region
FROM invoices as i
JOIN customers as c on i.customer_id = c.customer_number
WHERE amount < 1000

> EXERCISE 6 - Riportare l'elenco delle fatture (numero, importo, iva e data) con in aggiunta il nome del fornitore

SELECT 
i.invoice_number,
i.amount,
i.iva,
s.name AS supplier
FROM invoices as i
JOIN suppliers as s on i.supplier_number = s.supplier_number
